class Solution {
    public int searchInsert(int[] nums, int target) {
        
        int res=-1;
        
        for(int i=0;i<nums.length;i++)
        {
            if(nums[i]==target)
            {
                res=i;
            }
            if(nums[i]>target && res==-1)
            {
                res=i;
            }
        }
        
        if(res==-1)
        {
            return nums.length;
        }
        return res;
    }
}
